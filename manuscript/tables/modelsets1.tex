\multirow{3}{*}{$x \times z$} & $y=x_1+x_1z_1$ \\ & $y=x_1+x_1z_2$ \\ & $y=x_1+x_1z_1+x_1z_2$  \\& \\
$z \times z$ & $y=x_1+z_1z_2$ \\
